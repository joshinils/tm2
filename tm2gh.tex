    \subsection*{1.3 Stoffgesetz}

    Hooke'sches Gesetz
    \begin{equation}
      \boxed{
        \tag{1.8}
        \begin{split}
          \underbraceWrap[60pt]{E}{Elastizitätsmodul $\left[\frac{N}{mm^2}\right]$}
          =
          \frac{\overbrace{\sigma}^{\text{Spannung}\left[\frac{N}{mm^2}\right]}}

    \end{equation}

    Umgestellt nach Sigma, übliche Form:
    \begin{equation*}
        \begin{split}

    \end{equation*}

    \begin{equation}
      \boxed{
        \tag{1.9}
        \begin{split}
          \underbrace{\varepsilon}_{\text{Dehnung} \left[1\right]}

          \frac{\sigma}{E}

      }
    \end{equation}

    \begin{equation}

      }
    \end{equation}

    \begin{equation}

          \text{d}x

      }
    \end{equation}
     
    \begin{equation}
      \boxed{
        \tag{1.16}
        \begin{split}
          \Delta \ell

          \int_{0}^{\ell}

        \end{split}

    \end{equation}

    \begin{equation}
      \boxed{
        \tag{1.17}
        \begin{split}
          \Delta \ell

          \frac{F\ell}{EA}

        \end{split}
      }
    \end{equation}


    Für $\Delta T = 0$
    \begin{equation}
      \boxed{
        \tag{1.18}
        \begin{split}
          \Delta \ell

        \end{split}
      }
    \end{equation}

    Oder $F = 0$
    \begin{equation}
      \boxed{
        \tag{1.19}
        \begin{split}
          \Delta \ell

          \alpha_T \Delta T \ell
        \end{split}
      }
    \end{equation}
    
    \begin{equation}
      \boxed{
        \tag{1.20a}
        \begin{split}

            EA u'
